63 // So a problematic case is if this->inflectionPoint-2.0 < 0 than the shear rate is in the denominator and because it can
64 // be zero we may get nan/inf values. Therefore we have to check these cases and catch them
65 double epsilon = 10e-6;
66 
67 if ( abs(this->inflectionPoint-2.0) < epsilon ) // for a=2.0 we get gammaDot^{-0} which is 1
68 {
69  res = (-2.0)*(this->nu_0-this->nu_infty)*(this->fluid_index_n-1.0)*pow(this->characteristicTime, this->inflectionPoint)*pow(1.0+pow(this->characteristicTime*shearRate,this->inflectionPoint) , ((this->fluid_index_n-1.0-this->inflectionPoint)/this->inflectionPoint) );
70 }
71 else // in the other case we have to check that gammaDot is not zero because otherwise we get 1/0
72 {

75  shearRate = epsilon;
