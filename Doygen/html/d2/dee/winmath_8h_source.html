40 inline double acosh(double x)
41 {
42  if (x < 1)
43  x = -1; /* NaN */
44  else if (x == 1)
45  return 0;
46  else if (x > BIG_CRITERIA)
47  x += x;

49  x += sqrt((x + 1) * (x - 1));

54 {
55  int neg = x < 0;

57 
58  if (z < SMALL_CRITERIA) return x;
59  if (z < (1.0/(1<<DBL_MANT_DIG/5))) {
60  double x2 = z * z;
61  z *= 1 + x2 * (-1.0/6.0 + x2 * 3.0/40.0);
62  }
63  else if (z > BIG_CRITERIA) {
64  z = log(z + z);
65  }
66  else {
67  z = log(z + sqrt(z * z + 1));
